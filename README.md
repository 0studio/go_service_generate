# gostruct2sqlgenerator

 go build main.go -o gostruct2sqlgenerator

 ./gostruct2sqlgenerator example/example_1.go

 give a file example_1.go generate example_1_sub.go
 ```
 package example

// go build main.go -o gostruct2sqlgenerator
// ./gostruct2sqlgenerator example/example_1.go

import (
	"time"
)

type Hello struct {
	Id   int    `json:"id" mysql:"pk,defalut=1,type=bigint"` // id
	Name string `mysql:"pk,defalut='hello',name=helloName,type=varchar(10)"`
	Age  int
	T    time.Time `mysql:"type=int"`
	T2   time.Time
}
type World struct {
	Id int
}
 ```
 example_1_sub.go
 ```
 package example // do not edit this file ,this is generated by tools(https://github.com/0studio/gostruct2sqlgenerator)

import (
    "fmt"
    "time"
)

var ___importTime time.Time

func (this Hello) GenerateInsert() (sql string) {
    sql = fmt.Sprintf("insert into `Hello`(Id,helloName,Age,T,T2) values (%d,'%s',%d,%d,%s);",
        this.Id,
        this.Name,
        this.Age,
        this.T.Unix(),
        this.T2.Format("20060102150405"))
    return
}

func (this Hello) GenerateCreate() (sql string) {
    sql = "create table if not exists `Hello`(`Id` bigint NOT NULL DEFAULT 0,`helloName` varchar(10) NOT NULL DEFAULT '',`Age` int NOT NULL DEFAULT 0,`T` int NOT NULL DEFAULT 0,`T2` timestamp NOT NULL DEFAULT 0,primary key (Id,helloName));"
    return
}

func (this World) GenerateInsert() (sql string) {
    sql = fmt.Sprintf("insert into `World`(Id) values (%d);",
        this.Id)
    return
}

func (this World) GenerateCreate() (sql string) {
    sql = "create table if not exists `World`(`Id` int NOT NULL DEFAULT 0);"
    return
}
 ```
